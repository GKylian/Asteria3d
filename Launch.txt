#!/bin/bash

echo Deleting the previous problem file
rm "prob.h"

read -p 'Enter the problem file local directory (e.g. problems/otvortex.h): ' filename

if test -f "$filename"; then
    echo "Problem file $filename exists, creating copy prob.h for the code"
    cp "$filename" "prob.h"
else
    echo "The file $filename does not exist"
    exit 1
fi

#read -p 'Use MHD ? [y/n]: ' mhd

#if ["$mhd" == "y"]; then
#    echo "Configuring with MHD"
#elif ["$mhd" == "n"]; then
#    echo "Configuring with HD"
#else
#    echo "Invalid input"
#    exit 1
#fi

read -p 'Use HLLE, HLLC or HLLD ? ' riemann

if ["$riemann" == "HLLE"]; then
    echo "Configuring with the HLLE Riemann Solver"
elif ["$riemann" == "HLLC"]; then
    echo "Configuring with the HLLC Riemann Solver"
elif ["$riemann" == "HLLD"]; then
    echo "Configuring with the HLLD Riemann Solver"
else
    echo "Invalid input"
    exit 1
fi
